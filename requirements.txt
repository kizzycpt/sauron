# ────────────────────────────────────────────────────────────
# ARCH LINUX / SYSTEMD SYSTEMS
# ────────────────────────────────────────────────────────────

# Arch does not write UFW logs to /var/log/ufw.log by default.
# Instead, rsyslog routes them to /var/log/syslog.
# The dashboard auto-detects this, but the file may not be
# readable by your user.


# Check where your UFW logs are going:
#
#   sudo grep -r "UFW BLOCK" /var/log/ 2>/dev/null | head -3


# Check if the file is readable:
#
#   ls -la /var/log/syslog
#
# If it shows -rw-r--r-- you are fine.
# If it shows -rw-r----- (root/adm only), fix it:
#
#   sudo chmod o+r /var/log/syslog
#
#   or add yourself to the log group:
#
#   sudo usermod -aG adm $USER
#   newgrp adm


# Make sure rsyslog is running:
#
#   systemctl status rsyslog
#
# If not installed:
#
#   sudo pacman -S rsyslog
#   sudo systemctl enable --now rsyslog


# Make sure UFW logging is enabled:
#
#   sudo ufw logging on
#   sudo ufw status verbose
#
# Output should show "Logging: on"


# ────────────────────────────────────────────────────────────
# UBUNTU / DEBIAN / KALI
# ────────────────────────────────────────────────────────────

# UFW logs to /var/log/ufw.log directly.
# If the feed is empty:
#
#   sudo ufw logging on
#   sudo ufw reload
#   tail -f /var/log/ufw.log
#
# The last command lets you verify logs are flowing in
# real time before launching the dashboard.
